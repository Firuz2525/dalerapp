service cloud.firestore {
match /databases/{database}/documents {
    match /categories/{category} {
      allow read: if true; // Anyone can see categories
      allow write: if request.auth != null; // Only logged-in users can add/delete
    }
    match /brands/{brand} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
  match /databases/{database}/documents {
    
    // 1. Allow anyone to read products (for the storefront)
    match /products/{product} {
      allow read: if true;
      allow write: if request.auth != null; // Only logged-in users can add/delete
    }

    // 2. Metadata (Categories)
    match /metadata/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
